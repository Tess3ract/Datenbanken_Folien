% Beziehungstypen

\begin{frame}{Beziehungstypen}
    
    \begin{itemize}    
        \item<2-> Beziehungstypen beschreiben Zusammenhänge zwischen Entity-Typen
        \item<3-> Beispiel: Professoren \textbf{lesen} Vorlesungen
        \item<4-> Mathematisch:\\
              $\text{lesen} \subseteq \text{Professoren} \times \text{Vorlesungen}$
        \item<5-> Allgemein für $n$ Entitäten:\\
              $\text{R} \subseteq E_1 \times E_2 \times \cdots E_n$
    \end{itemize}
    
    \vspace{0.3cm}
    
    \centering
    \only<1->{    
            \begin{tikzpicture}    
                
                % Entitäten
                \node[entity] (prof) at (0,0) {Professoren};
                \node[entity] (vl) at (6,0) {Vorlesungen};
                
                % Beziehung
                \node[relation] (lesen) at (3,0) {lesen};
                
                % Attribute Professoren
                \node[attribute] (persnr) at (0,1.3) {\underline{PersNr}};
                \node[attribute] (pname) at (0,-1.3) {Name};
                
                % Attribute Vorlesungen
                \node[attribute] (vorlnr) at (6,1.3) {\underline{VorlNr}};
                \node[attribute] (raum) at (6,-1.3) {Raum};
                
                % Kanten
                \draw (prof) -- (lesen);
                \draw (lesen) -- (vl);
                
                \draw (prof) -- (persnr);
                \draw (prof) -- (pname);
                
                \draw (vl) -- (vorlnr);
                \draw (vl) -- (raum);
                
            \end{tikzpicture}
        }
\end{frame}
