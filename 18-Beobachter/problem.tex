\note{
	Wenn wir eine Tageszeitung lesen möchten, können wir jeden Tag zum Kiosk
	gehen und die Zeitung kaufen.
}

\begin{frame}{Verlag versendet Zeitung an jeden}
	\begin{center}
 		\includegraphics[scale=0.3]{\BILDERPFAD/problem3.png}
 	\end{center}
\end{frame}
\note{
	Alternativ kann der Verlag die Zeitung an jedes Haus versenden. Das wird 
	einigen wenigen Menschen gefallen, aber die meisten wird es nerven.
}

\begin{frame}{Verlag versendet Zeitung nur an Abonnenten}
	\begin{center}
 		\includegraphics[scale=0.3]{\BILDERPFAD/problem4.png}
 	\end{center}
\end{frame}
\note{
	Die beste Lösung ist jedoch, dass der Verlag die Zeitung nur an diejenigen
	verwendet, die die Zeitung auch lesen möchten. Die Leser abonnieren also
	die Zeitung und alle, die sich nicht dafür interessieren, abonnieren nicht.
	Die Abonnenten werden nun jeden Tag benachrichtigt, wenn es eine neue 
	Zeitung gibt.
}
