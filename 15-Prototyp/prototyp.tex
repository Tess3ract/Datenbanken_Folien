\begin{frame}{Prototyp}
	\begin{block}{Zweck}
 		Bestimme die Arten zu erzeugender Objekte durch die Verwendung eines prototypischen Exemplars und erzeuge neue Objekte durch Kopieren dieses Prototypen.
 	\end{block}
\end{frame}

\begin{frame}[fragile]{Prototyp}
	\begin{exampleblock}{Shape}
 		\begin{lstlisting}
public abstract class Shape implements Cloneable {
	private double x;
	private double y;

	public Shape(double x, double y) {
		this.x = x;
		this.y = y;
	}
	
	@Override
	public Object clone() throws CloneNotSupportedException {
		return super.clone();
	}
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{KonkreterPrototyp 1}
	\begin{exampleblock}{Circle}
 		\begin{lstlisting}
public class Circle extends Shape {
	private double radius;
	
	public Circle(double x, double y, double radius) {
		super(x, y);
		this.radius = radius;
	}

	@Override
	public Circle clone() throws CloneNotSupportedException {
		return (Circle)super.clone();
	}
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{KonkreterPrototyp 2}
	\begin{exampleblock}{Rectangle}
 		\begin{lstlisting}
public class Rectangle extends Shape {
	private double width;
	private double height;

	public Rectangle(double x, double y, double height,
            double width) {
		super(x, y);
		this.height = height;
		this.width = width;
	}
	
	@Override
	public Rectangle clone() throws CloneNotSupportedException {
		return (Rectangle)super.clone();
	}
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Klient}

\note{
	Im Beispiel haben wir nur einfache Datentypen kopiert. Aber oft hat man 
	Objektbäume, bei denen man überlegen muss, ob das Kopieren von Referenzen
	reicht (flache Kopie) oder ob die kompletten Objekte kopiert werden müssen
	(tiefe Kopie).
}

\note{
	Ein etwas realistischeres Beispiel: Wir möchten gerne eine Software für die Verwaltung von
	Prüfungen. Dabei werden die Prüfungen in einer xml-Datei gespeichert. Die Prüfungen werden
	aus der xml-Datei gelesen und neue Prüfungen können als Klon vorhandener Prüfungen erzeugt
	werden. In diesem Fall werden die Java-spezifischen Möglichkeiten nicht verwendet, da dies 
	nicht von jeder Programmiersprache in dieser Form unterstützt wird. 
}

\begin{frame}[fragile]{Prototyp}
	\begin{exampleblock}{Examination}
 		\begin{lstlisting}
public abstract class Examination {
	private String date;
	private Room room; 
	
	public Examination(String date, Room room) {
		this.date = date;
		this.room = room;
	}
	
	public abstract Examination clone();
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Prototyp - Flaches Kopieren}
	\begin{exampleblock}{Project}
 		\begin{lstlisting}
public class Project extends Examination {
	private String title;

	public Project(String date, String title, Room room) {
		super(date, room);
		this.title = title;
	}
	
	@Override
	public Project clone() {
		return new Project(getDate(), title, getRoom());
	}
	
	public String getTitle() {
		return title;
	}
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Konkreter Prototyp 2 - Tiefes Kopieren}
	\begin{exampleblock}{WrittenExamination}
        \begin{lstlisting}
public class WrittenExamination extends Examination {
	private String timeStart;
	private String timeEnd;
	
	public WrittenExamination(String date, 
			String timeStart, String timeEnd, Room room) {
		super(date, room);
		this.timeStart = timeStart;
		this.timeEnd = timeEnd;
	}
	
	@Override
	public WrittenExamination clone() {
		return new WrittenExamination(getDate(),  
				timeStart, timeEnd, getRoom().clone());
	}
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}
