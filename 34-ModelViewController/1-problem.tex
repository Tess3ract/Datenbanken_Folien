}

\begin{frame}{Anforderungen an grafische Oberflächen}
	\begin{itemize}
		\item Darstellung derselben Information in unterschiedlichen Fenstern
		\item Darstellung derselben Information auf verschiedene Arten
		\item Sofortige Widerspiegelung von Datenänderungen
		\item Leichte Anpassbarkeit der Benutzeroberfläche (sogar zur Laufzeit)
		\item Änderung des Erscheinungsbildes ohne Auswirkungen auf den Anwendungskern
	\end{itemize}
\end{frame}
