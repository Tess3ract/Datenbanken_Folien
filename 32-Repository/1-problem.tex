\writeSectionFrame{\presentationTitle}{Problem}
\begin{frame}{Problem: Direkter Datenzugriff}
    \begin{itemize}
        \item Geschäftslogik direkt auf Datenbank, Filesystem oder externen Service → stark gekoppelt
        \item Viele Klassen enthalten SQL, API-Aufrufe oder ORM-Code
        \item Änderungen an der Datenquelle erfordern Änderungen überall
        \item Testbarkeit wird erschwert: schwer isolierbar
    \end{itemize}
    \pause
    \begin{alertblock}{Beispiel}
        playerRepository.save(player);
        playerRepository.findById(42);
    \end{alertblock}
\end{frame}

\begin{frame}{Problem: Testbarkeit}
    \begin{itemize}
        \item Ohne Repository muss jede Geschäftslogik echte Datenbanken ansprechen
        \item Unit-Tests schwierig $\rightarrow$ Integrationstests notwendig
        \item Duplizierter Datenzugriff in mehreren Klassen
        \item Schwer wiederverwendbare, saubere APIs
    \end{itemize}
\end{frame}

\begin{frame}{Motivation: Entkopplung von der Datenquelle}
    \begin{itemize}
        \item Geschäftslogik sollte nur wissen: Ich möchte Objekte speichern, laden, suchen
        \item Repository kapselt alle Details der Persistenz
        \item Vorteile:
        \begin{itemize}
            \item Einheitliche Schnittstelle für Datenzugriff
            \item Austausch der Implementierung möglich (DB, File, Mock)
            \item Erhöht Testbarkeit
        \end{itemize}
    \end{itemize}
\end{frame}
