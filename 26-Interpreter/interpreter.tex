\note{
	Das Interpreter-Muster beschreibt den Aufbau einer Grammatik, damit sie von einem Interpreter
	ausgewertet werden kann. Außerdem muss der Ausdruck geparst werden. Der Parser ist jedoch nicht
	Teil des Musters. Es geht bei dem Muster darum, einen abstrakten Syntax-Baum zu repräsentieren
	und diesen zu interpretieren.
}

\begin{frame}[fragile]{TerminalExpression}
	\begin{exampleblock}{TerminalExpression.java}
 		\begin{lstlisting}
public class TerminalExpression implements Expression {
	private String searchTerm;

	@Override
	public List<Book> interpret(BookContentSearchContext 
            interpreterBookContext) {
		List<Book> foundBooks = new ArrayList<>();
		List<Book> books = interpreterBookContext.getBooks();
		
		for (Book book: books)  {
			String content = book.getContent();
			if (content.contains(searchTerm)) {
				foundBooks.add(book);
			}
		}
		
		return foundBooks;
	}
}
 		\end{lstlisting}
 	\end{exampleblock}
