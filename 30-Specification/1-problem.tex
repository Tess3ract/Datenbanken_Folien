\writeSectionFrame{\presentationTitle}{Problem}
\begin{frame}{Motivation}
	\begin{itemize}
	  \item Geschäftsregeln sind oft im Code verteilt
	  \item Bedingungen tauchen mehrfach auf
	  \item Änderungen sind fehleranfällig
	\end{itemize}
	\pause
	\begin{alertblock}{Typisches Problem}
		Komplexe If-Bedingungen werden unübersichtlich und schwer wartbar.
	\end{alertblock}
\end{frame}


\begin{frame}[fragile]{Ohne Specification}
	\begin{exampleblock}{Beispiel}
		\begin{lstlisting}
if (player.getLevel() >= 10
    && player.getGold() >= 100
    && !player.isCursed()) {
    // Spieler darf Gegenstand kaufen
}
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}{Probleme}
	\begin{itemize}
	  \item Logik ist schlecht lesbar
	  \item Schwer wiederverwendbar
	  \item Änderungen erfordern viele Anpassungen
	\end{itemize}
\end{frame}
