\begin{frame}{Mediator}
	\begin{block}{Zweck}
 		Definiere ein Objekt, welches das Zusammenspiel einer Menge von Objekten in sich kapselt. Vermittler fördern lose Kopplung, indem sie Objekte davon abhalten, aufeinander explizit Bezug zu nehmen. Sie ermöglichen es, das Zusammenspiel der Objekte von ihnen unabhängig zu variieren.
 	\end{block}
\end{frame}

\note{
	Wir möchten im ersten Beispiel gerne eine VOIP-Telefonanlage simulieren. Es gibt 
	verschiedene Arten von Teilnehmern. Jederzeit können neue Teilnehmer
	an den Switch angeschlossen und bereits existierende entfernt werden. VOIP-Geräte
	sind weitestgehend autonom, können z.B. selbstständig Verbindungen aufbauen. Der Switch
	erledigt Verwaltungsarbeit wie z.B. das Ermitteln eines Teilnehmers anhand der Telefonnummer
	oder die Erstellung von Protokollen oder Konferenzen. 
}

\begin{frame}[fragile]{Konkreter Vermittler}
	\begin{exampleblock}{Telefonanlage}
 		\begin{lstlisting}
public class TelefonAnlage implements Switch {
	@Override
	public void verbinden(Teilnehmer teilnehmer, 
			Teilnehmer gegenstelle) {
		if (gegenstelle.istBesetzt()) {
			throw new SwitchException("...");
		}
		teilnehmer.verbinden(gegenstelle);
	}
	
	@Override
	public void konferenzEroeffnen(Teilnehmer teilnehmer, 
			Teilnehmer [] gegenstellen) {
		for (Teilnehmer gegenstelle: gegenstellen) {
			verbinden(teilnehmer, gegenstelle);
		}
	}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

public class FlightController implements Mediator {
