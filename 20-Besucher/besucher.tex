\begin{frame}{Besucher}
	\begin{block}{Zweck}
 		Kapsle eine auf den Elementen einer Objektstruktur auszuführende Operation als ein Objekt. Das Besuchermuster ermöglicht es, eine neue Operation zu definieren, ohne die Klasse der von ihr bearbeiteten Elemente zu verändern.
 	\end{block}
\end{frame}

\begin{frame}{Allgemeines Klassendiagramm}
	\begin{center}
 		\includegraphics[scale=0.3]{\BILDERPFAD/besucher1.png}
 	\end{center}
\end{frame}

\note{
	Besucher hilft dabei, die Klassenhierarchie von ihren Operationen
	zu trennen. Die Operationen werden in eine zweite Hierarchie 
	ausgelagert und statt die Operationen direkt auszuführen, wird ein Objekt
	von der Operation erstellt und dieses wird dann an die fachliche Hierarchie
	übergeben, die diese Operation dann aufruft.
}

\begin{frame}[fragile]{Element}
	\begin{exampleblock}{Artikel}
 		\begin{lstlisting}
public abstract class Artikel {
	private String titel;
	private int bestand;
	private double preis;
	
	public abstract void nimmEntgegen(Besucher besucher);

	public void erhoeheStueckzahl(int menge) {
		bestand = bestand + menge;
	}
} 		    
 		\end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Konkretes Element}
	\begin{exampleblock}{Film}
 		\begin{lstlisting}
public class Film extends Artikel {
	private final int laenge;
	private final String regisseur;
	
	public Film(String titel, int bestand, double preis, 
            int laenge, String regisseur) {
		super(titel, bestand, preis);
		this.laenge = laenge;
		this.regisseur = regisseur;
	}
	
	@Override
	public void nimmEntgegen(Besucher besucher) {
		besucher.besucheFilm(this);
	}
}
 		\end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Objektstruktur}
	\begin{exampleblock}{Warenkorb}
 		\begin{lstlisting}
public class Warenkorb {
	private List<Artikel> artikel;
	
	public double berechneGesamtpreis() {
		BerechneGesamtpreisBesucher besucher = new 
				BerechneGesamtpreisBesucher();
		artikel.forEach(einArtikel -> einArtikel.
            nimmEntgegen(besucher));
		return besucher.getPreis();
	}
	
	public void druckeAlsPdf() {
		PdfBesucher pdfBesucher = new PdfBesucher();
		artikel.forEach(einArtikel -> einArtikel.
            nimmEntgegen(pdfBesucher));
	}
} 		    
 		\end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Konkreter Besucher 2}

public class PageCountVisitor implements DocumentVisitor {
	private int totalPageCount = 0;

	@Override
	public void visitPdfDocument(PdfDocument pdf) {
		totalPageCount += pdf.getNumberOfPages();
	}

	@Override
	public void visitWordDocument(WordDocument word) {
		totalPageCount += (word.getContent().length() / 1000) + 1;
	}

	public int getTotalPageCount() {
		return totalPageCount;
	}
}

\begin{frame}[fragile]{Konkreter Besucher 1}
	\begin{exampleblock}{AnnualReturnVisitor}
 		\begin{lstlisting}
public class AnnualReturnVisitor implements InvestmentVisitor {
	private double totalAnnualReturn = 0;

	@Override
	public void visitStock(Stock stock) {
		totalAnnualReturn += stock.getMarketValue() * 
            0.05 + stock.getDividends();
	}

	@Override
	public void visitBond(Bond bond) {
		totalAnnualReturn += bond.getFaceValue() * 
            bond.getInterestRate();
	}
