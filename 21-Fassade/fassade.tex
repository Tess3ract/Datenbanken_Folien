\begin{frame}{Fassade}
	\begin{block}{Zweck}
 		Biete eine einheitliche Schnittstelle zu einer Menge von Schnittstellen eines Subsystems. Die Fassadenklasse definiert eine abstrakte Schnittstelle, welche die Verwendung des Subsystems vereinfacht.
 	\end{block}
\end{frame}

\note{
	Die Fassade kann man mit einem Call-Center vergleichen. Wenn ich telefonisch eine 
	Bestellung aufgebe, lande ich in einem Call-Center. Irgend jemand aus dieser
	Fassade nimmt die Bestellung entgegen. Im Hintergrund wird dann ein riesiges
	Subsystem gestartet, aber davon bekomme ich als Kunde nichts mit. Das Call-Center ist 
	eine Fassade, die das ganze System vor mir versteckt. 
}

\note{
	Die System von Schleckermann und BestellFix sollen jetzt miteinander verbunden werden. Die Bestellnummern sind unterschiedlich aufgebaut, die von Schleckermann sind einfache Ganzzahlen, während die von BestellFix mit BF beginnen, und auch die Namen der Untermodule sind unterschiedlich. Die Fassade soll eine Bestellung anhand der Bestellnummer einem Sub-System zuordnen und die entsprechende Methode aufrufen, um eine Bestellung
	zu starten.
}

\begin{frame}[fragile]{Fassade}
	\begin{exampleblock}{VersandhandelFassade}
 		\begin{lstlisting}
public class VersandhandelFassade {
	private Bestellannahme bestellannahme;
	private Bestelleingang bestelleingang;
	
	public VersandhandelFassade() {
		bestelleingang = new Bestelleingang();
	}
	
	public void starteBestellung(String bestellNummer) {
		if (bestellNummer.startsWith("BF")) {
			bestelleingang.setBestellNr(bestellNummer);
			bestelleingang.starteBestellung();
		} else {
			int bestellnummer = Integer.parseInt(bestellNummer);
			bestellannahme = new Bestellannahme(bestellnummer);
		}
	}
} 		    
 		\end{lstlisting}
 	\end{exampleblock}
\end{frame}

\note{
	Im Beispiel sollen mehrere Bibliotheken zum Auslesen von Ontologien simuliert werden: OWLApi und
	Jena. Beide enthalten unterschiedliche Klassen und haben unterschiedliche Methoden. Die beiden
	Systeme sollen hinter einer Fassade versteckt werden. Zusätzlich soll die Fassade die Ergebnistypen
	in neue Klassen überführen, die unabhängig vom verwendeten Framework sind. So kann der Client
	unabhängig gehalten werden vom konkreten Framework, die Frameworks werden austauschbar und es ist 
	möglich, dass der Client für einige Anfragen die OWLApi und für andere Jena verwendet. So können
	die Stärken des jeweiligen Frameworks genutzt werden, ohne dass Änderungen am Client vorgenommen
	werden müssen.
}

public List<MyOntologyClass> readAllOntologies() {
    return readAllOntologiesFromJena();

private List<MyOntologyClass> readAllOntologiesFromOwlApi() {
    List<OwlClass> owlClasses = owlOntology.
        loadAllOntologyClasses();
    List<MyOntologyClass> myOntologyClasses = new ArrayList<>();
    
    owlClasses.forEach(owlClass -> { 
        final String iri = owlClass.getIri().toString();
        myOntologyClasses.add(new MyOntologyClass(iri));
    });
    
    return myOntologyClasses;
}
