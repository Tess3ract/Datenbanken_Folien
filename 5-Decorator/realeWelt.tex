\begin{frame}{Java I/O}
	\begin{itemize}
 		\item Kernkomponenten: InputStream, OutputStream, Reader, Writer
        \item Konkrete Komponenten: FileInputStream, FileOutputStream, BufferedReader, BufferedWriter, etc.
 		\item Dekoratoren: Klassen wie BufferedInputStream, DataInputStream, GZIPOutputStream, PrintWriter
 	\end{itemize}
\end{frame}
\note{
    In Java I/O werden die Dekoratoren verwendet, um zusätzliche Funktionalitäten wie Pufferung, Datenumwandlung, Komprimierung usw. einzubringen. 
}

\begin{frame}[fragile]{Unser neuer Dekorierer}
	\begin{exampleblock}{LowerCaseInputStream.java}
		\begin{lstlisting}
public class LowerCaseInputStream extends FilterInputStream {
	public LowerCaseInputStream(InputStream in) {
		super(in);
	}
	
	public int read() throws IOException {
		int c = super.read();
		if (c == -1) {
			return c; 
		} else {
			return Character.toLowerCase((char)c);
		}
	}
}
		\end{lstlisting}
	 \end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Servlet-Filter in Jakarta EE}
	\begin{exampleblock}{Beispiel}
		\begin{lstlisting}
public class CompressionFilter implements Filter {
    @Override
    public void doFilter(ServletRequest request, 
                ServletResponse response, FilterChain chain) 
            throws IOException, ServletException {
        HttpServletResponse httpResp = (HttpServletResponse) 
            response;
        GZIPResponseWrapper wrappedResponse = 
            new GZIPResponseWrapper(httpResp);
        chain.doFilter(request, wrappedResponse);
        wrappedResponse.finishResponse();
    }
}
        \end{lstlisting}
	\end{exampleblock}
\end{frame}
\note{
    In Webanwendungen, die mit Java EE (jetzt Jakarta EE) entwickelt wurden, werden Servlet-Filter verwendet, um Anfragen und Antworten zu modifizieren. Kernkomponenten: HttpServletRequest, HttpServletResponse, Dekoratoren sind Filter wie CompressionFilter, LoggingFilter, AuthenticationFilter, die zusätzliche Funktionen wie Komprimierung, Protokollierung und Authentifizierung hinzufügen. Ein CompressionFilter komprimiert eine Antwort, bevor sie an den Client gesendet wird.
}
