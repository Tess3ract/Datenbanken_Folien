\writeSectionFrame{\presentationTitle}{Das Command-Muster}
\begin{frame}{Steckbrief}
	\begin{itemize}
 		\item Deutscher Name: Befehl
 		\item Alternativer deutscher Name: Kommando
 		\item Englischer Name: Command
 		\item Gruppe: Verhaltensmuster
 	\end{itemize}
\end{frame}

\begin{frame}{Befehl}
	\begin{block}{Zweck}
 		Eine Anfrage als ein Objekt kapseln. Dies ermöglicht es, Klienten mit verschiedenen Anfragen zu parametrisieren, Operationen in eine Warteschlange zu stellen, ein Logbuch zu führen und Operationen rückgängig zu machen.
 	\end{block}
\end{frame}

\begin{frame}{Beispiel aus dem realen Leben}
	\myImageWithSource{0.5}{\BILDERPFAD/command1.png}{https://refactoring.guru/design-patterns/command}
\end{frame}
\note{
    Das Befehlsmuster dürfte eines der am schwersten verständlichen sein. Deswegen näheren wir uns dem Thema mit einem Beispiel aus dem realen Leben. Sie laufen durch die Stadt und haben Hunger, weil Sie schon den ganzen Tag unterwegs waren. Sie gehen in ein Restaurant und werden von einem freundlichen Kellner bedient. Er nimmt Ihre Bestellung auf und schreibt diese auf einen Zettel. Diesen Zettel gibt er in der Küche ab. Der Koch bearbeitet die Zettel nacheinander. Die gekochten Gerichte stellt er zusammen mit dem Bestellzettel zum Abholen bereit. Der Kellner sieht, dass das Gericht fertig gekocht, kann es dem Tisch zuordnen und bringt es zu Ihnen. Der Zettel ist das Befehlsobjekt, das in einer Warteschlange verbleibt, bis der Koch sich darum kümmert. So kann der Koch sich darum kümmern zu kochen statt zu den Tischen zu laufen und sich mit den Bestelldetails auseinanderzusetzen.
}

\begin{frame}{Schichtenarchitektur}
	\myImageWithSource{0.6}{\BILDERPFAD/command2.png}{https://refactoring.guru/design-patterns/command}
\end{frame}
\note{
    Gutes Softwaredesign basiert auf Trennung der Belange und häufig wird dazu die Schichtenarchitektur angewendet. Wenn wir als Beispiel eine grafische Benutzeroberfläche nehmen, dann haben wir mehrere Elemente in der Benutzeroberfläche, die ihre Daten an die Geschäftslogik senden. Diese übernimmt deren Verarbeitung und führt die Logik aus. Die Komponenten der GUI senden also eine Anfrage an die Geschäftslogik. Das Problem ist, dass in diesem Beispiel alle drei Komponenten das gleiche machen und prinzipiell die gleiche Anfrage senden.
}

\begin{frame}{Schichtenarchitektur mit Befehlsobjekt}
	\myImageWithSource{0.6}{\BILDERPFAD/command3.png}{https://refactoring.guru/design-patterns/command}
\end{frame}
\note{
    Das Befehlmuster schreibt ein weiteres Objekt vor, das Befehlsobjekt. Die GUI sollte die Geschäftslogik nicht sofort aufrufen, sondern soll alle Daten extrahieren. Dazu gehören neben der Parameterliste auch der Name der Methode. Diese werden an das Befehlsobjekt übergeben, das nur eine einzige Methode hat, die den Request auslöst. Dieses Objekt dient als Verbindung zwischen verschiedenen Elementen der GUI und der Geschäftslogik. Die GUI braucht nicht zu wissen, welches Element der Geschäftslogik die Anfrage erhält und wie sie verarbeitet wird. Die GUI stößt nur das Kommando an und dieses kümmert sich um alle Details. 
}

\begin{frame}{Befehlsobjekt mit Interface}
	\myImageWithSource{0.6}{\BILDERPFAD/command4.png}{https://refactoring.guru/design-patterns/command}
\end{frame}
\note{
    Der nächste Schritt ist natürlich, ein Interface für das Befehlsobjekt zu erschaffen, da alle Befehlsobjekte dieselbe Schnittstelle implementieren. Normalerweise hat es nur eine einzige parameterlose Methode. Dieses Interface erlaubt es, verschiedene Befehle mit demselben Sender auszuführen, ohne den Sender an eine konkrete Klasse zu koppeln. Als kleiner Bonus ist es auf diese Art sogar einfach möglich, Befehlsobjekte auszutauschen und somit das Verhalten eines Senders zur Laufzeit zu ändern. Eine Frage ist noch, wie das Befehlsobjekt an die Parameter kommt, wenn die Methode parameterlos ist und einsetzbar für jede Art von Request. Das sehen wir uns gleich an. Wir implementieren also Befehlsobjekte für jede mögliche Operation und stecken eine Referenz auf die Objekte in die GUI-Komponenten. Somit sind sie mit einem Befehl verbunden, der ausgeführt wird, wenn der Benutzer mit dem Element interagiert. Somit werden die Befehlsobjekte zu einer Zwischenschicht, die für eine lose Kopplung sorgt zwischen GUI und Geschäftslogik. Und da wir mit einem Interface arbeiten, ist das konkrete Befehlsobjekt und damit das Verhalten austauschbar. 
}

\note{
	Ein Klient erzeugt das Befehl-Objekt, das aus einem Satz von Aktionen auf einem 
	Empfänger besteht. Das Befehl-Objekt bietet die einzige Methode ausführen(). Diese 
	kapselt die Aktionen und kann aufgerufen werden, um die Methoden auf dem Empfänger
	auszuführen. Der Klient ruft setBefehl() auf einem Aufrufer-Objekt auf mit dem
	Befehl-Objekt als Übergabe. Dieses Objekt wird im Aufrufer gespeichert, bis es
	benötigt wird. Irgendwann in der Zukunft fordert der Klient den Aufrufer auf, 
	den Befehl auszuführen. Und dann werden auch die Aktionen auf dem Empfänger
	ausgeführt. 
}

\note{
	Der Aufrufer initiiert Anfragen. Diese Klasse braucht ein Feld zur Speicherung einer Referenz auf Befehlsobjekte. Der Aufrufer stößt das Kommando an statt direkt den Empfänger aufzurufen. Der Aufrufer erzeugt die Befehlsobjekte nicht selbst, sondern erhält sie normalerweise vom Klienten über den Konstruktor. Das Interface Befehl enthält eine Methode zum Auführen der Operation. Konkrete Befehle implementieren das Interface mit einem konkreten Verhalten. Dabei soll ein konkretes Kommando die Arbeit nicht selbst machen, sondern den Aufruf an ein Objekt der Geschäftslogik weiterleiten. Parameter, die zur Ausführung einer Methode benötigt werden, können als Zustand der Befehle in Attributen gespeichert werden. Befehlsobjekte sollten unveränderlich sein. Der Empfänger enthält die Geschäftslogik und somit kann es sich dabei um beinahe jedes beliebige Objekt handeln. Die Befehlsobjekte kümmern sich im Allgemeinen nur um die Details, wie ein Befehl an einen Empfänger übertragen wird, während der Empfänger die wirkliche Arbeit ausführt. Der Klient erzeugt reale Befehlsobjekte und übergibt ihnen seine Parameter und eine Referenz auf den Empfänger im Konstruktor. Der Aufrufer kann auch mehrere Referenzen auf Befehle enthalten. Dann kann man z.B. eine Undo-Funktionalität realisieren.
}

public class ComplexRemoteControl {
	private static final int COMMAND_COUNT = 7;
	
	private Command[] onCommands;
	private Command[] offCommands;
	
	public ComplexRemoteControl() {
		onCommands = new Command[COMMAND_COUNT];
		offCommands = new Command[COMMAND_COUNT];
		
		for (int i = 0; i < COMMAND_COUNT; i++) {
			onCommands[i] = new NoCommand();
			offCommands[i] = new NoCommand();
		}
	}

\begin{frame}[fragile]{Aufrufer}
	\begin{exampleblock}{EditorInvoker.java}
 		\begin{lstlisting}
public class EditorInvoker {
	private Stack<TextEditorCommand> executed = new Stack<>();
    private Stack<TextEditorCommand> history = new Stack<>();

    public void executeCommand(TextEditorCommand command) {
        command.execute();
        
        if (!(command instanceof CopyCommand)) { 
            history.push(command);
        }
        
        if (command instanceof WriteCommand) {
        	executed.add(command);
        }
    }
 		\end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Aufrufer}
	\begin{exampleblock}{EditorInvoker.java}
 		\begin{lstlisting}
public void repeatCommand() {
    TextEditorCommand command = executed.peek();
    command.execute();
}

public void undo() {
    if (!history.isEmpty()) {
        TextEditorCommand command = history.pop();
        executed.push(command);
        command.undo();
    }
}
 		\end{lstlisting}
 	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Konkretes Kommando}
	\begin{exampleblock}{WriteCommand.java}
 		\begin{lstlisting}
public class WriteCommand implements TextEditorCommand {
    private TextEditor editor;
    private String text;

    public WriteCommand(TextEditor editor, String text) {
        this.editor = editor;
        this.text = text;
    }

    public void execute() {
        editor.write(text);
    }

    public void undo() {
        editor.delete(editor.getText().length() - text.length());
    }
}
 		\end{lstlisting}
 	\end{exampleblock}

    public void write(String newText) {

    public void write(int index, String newText) {
    	text.insert(index, newText);

    public void delete(String deletedText) {
    	int start = text.indexOf(deletedText);

    		text.delete(start, start + deletedText.length());
    	}

public class ChessGame {
    private List<Command> commandHistory;
    
    public void performMove(Command command) {
        command.execute();
        commandHistory.add(command);
    }

    public void undoLastMove() {
        if (!commandHistory.isEmpty()) {
            Command lastCommand = commandHistory
                .remove(commandHistory.size() - 1);
            lastCommand.undo();
        } else {
            System.out.println("No move to undo");
        }
    }
}
