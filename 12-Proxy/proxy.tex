\begin{frame}{Proxy}
	\begin{block}{Zweck}
 		Kontrolliere den Zugriff auf ein Objekt mithilfe eines vorgelagerten Stellvertreterobjekts.
 	\end{block}
\end{frame}

\note{
	In bestimmten Fällen soll das Versenden der Emails jedoch noch einmal separat
	protokolliert werden, bspw. wenn Rechnungen oder Mahnungen versendet werden. 
	Außerdem sollen bestimmte Emails nur zu bestimmten Zeiten verwendet werden, bspw. 
	nur während den üblichen Bürozeiten und nicht in der Nacht. Wir haben bereits
	eine Klasse Mailer, aber diese möchten wir nicht verändern, denn die neuen Funktionen
	sollen ja nicht immer eingesetzt werden. 
}

\begin{frame}[fragile]{Proxy}
	\begin{exampleblock}{TimerMailerProxy.java}
        \begin{lstlisting}
@Override
public void sendeEmail(String empfaenger, 
        String sender, String subject, String body) {
    int hour = random.nextInt(24);
    
    if (hour < FIRST_SEND_HOUR || hour > LAST_SEND_HOUR) {
        MailRequest mailRequest = new MailRequest(sender,
            empfaenger, body, subject);
        messageQueue.add(mailRequest);
    } else {
        mailer.sendeEmail(empfaenger, sender, subject, body);
        if (messageQueue.size() > 0) {
            for (MailRequest mailRequest: messageQueue) {
                mailer.sendeEmail(/*...*/);
            }
            messageQueue.clear();
        }
    } 
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}

public class CommandExecutorImpl implements CommandExecutor {
	private VirtualFileSystem virtualFileSystem;
	
	@Override
	public void executeCommand(String command) {
		if (command.startsWith("rm")) {
			String[] parts = command.split(" ");
			String filename = parts[1];
			virtualFileSystem.rm(filename);
		} else if (command.startsWith("ls")) {
			virtualFileSystem.ls().forEach(file -> System.out.println(file));
		} else if (command.startsWith("cat")) {
			// ...;
		}
	}
}
        \end{lstlisting}
 	\end{exampleblock}
\end{frame}
