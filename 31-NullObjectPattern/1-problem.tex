\writeSectionFrame{\presentationTitle}{Problem}
\begin{frame}{Problem: null in Code}
    \begin{itemize}
        \item Oft müssen Objekte auf null geprüft werden
        \item Viele if-Abfragen: \texttt{if (obj != null) \{ ... \}}
        \item Fehleranfällig: NullPointerExceptions
        \item Code wird unübersichtlich und schwer testbar
    \end{itemize}
    \pause
    \begin{alertblock}{Beispiel}
    	// riskant, wenn player.getWeapon() null ist
        player.getWeapon().attack(); 
    \end{alertblock}
\end{frame}

\begin{frame}{Mögliche Lösungen}
	\begin{block}{In Sprachen eingebaute Lösungen}
	    \begin{itemize}
	        \item Optional in Java
	        \item Nicht-nullbare Objekte in Kotlin (?-Operator)
	    \end{itemize}
    \end{block}
    \pause
    \begin{alertblock}{Aber \ldots}
   		was ist, wenn im Null-Fall eine andere Aktion durchgeführt werden soll?
    \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Ausführen einer anderen Aktion}
	\begin{exampleblock}{Standardlösung in Java}
		\begin{lstlisting}
if (player.getWeapon() == null) {
	store.buyWeapon(player);
}		
		\end{lstlisting}
	\end{exampleblock}
\end{frame}
