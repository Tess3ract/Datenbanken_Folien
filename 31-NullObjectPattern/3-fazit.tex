\begin{frame}{Null Object Pattern}
    \begin{itemize}
        \item Explizite Objekte statt null
        \item Reduziert Fehleranfälligkeit
        \item Lesbarer und wartbarer Code
        \item Passt gut zu modernen Patterns:
        \begin{itemize}
            \item Fluent API
            \item Specification
            \item Dependency Injection
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Vorteile des Null Object Pattern}
    \begin{itemize}
        \item Keine null-Prüfungen nötig
        \item Vereinfachter, lesbarer Code
        \item Einheitliches Verhalten
        \item Testbarkeit verbessert
        \item Kombinierbar mit Fluent / Specification / DI
    \end{itemize}
\end{frame}

\begin{frame}{Nachteile / Grenzen}
    \begin{itemize}
        \item Zusätzliche Klassen / Objekte
        \item Nicht jede null-Situation lässt sich sinnvoll abbilden
        \item Kann zu stillem Fehler führen, wenn Objekt nichts tut
    \end{itemize}
\end{frame}
