\writeSectionFrame{\presentationTitle}{Das Problem}
\begin{frame}{Motivation}
    \begin{itemize}
        \item Objekte besitzen internen Zustand
        \item Dieser Zustand kann sich zur Laufzeit ändern
        \item Änderungen sind oft schwer nachvollziehbar
    \end{itemize}
    \pause
    \begin{alertblock}{Typische Probleme}
        \begin{itemize}
            \item Versteckte Seiteneffekte
            \item Schwierige Fehlersuche
            \item Komplizierte Tests
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Mutables Objekt – Beispiel}
    \begin{exampleblock}{Monster.java}
        \begin{lstlisting}
public class Monster {
    private int health;

    public Monster(int health) {
        this.health = health;
    }

    public void takeDamage(int damage) {
        health -= damage;
    }
}
        \end{lstlisting}
    \end{exampleblock}
\end{frame}

\begin{frame}{Problem}
    \begin{itemize}
        \item Objekt verändert seinen internen Zustand
        \item Jede Methode kann Seiteneffekte haben
        \item Reihenfolge von Methodenaufrufen ist relevant
    \end{itemize}
\end{frame}
